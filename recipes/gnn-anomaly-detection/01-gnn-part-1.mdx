## Why Graph Neural Networks for Stock Anomaly Detection

Traditional anomaly detection methods treat each stock as an independent time series. Isolation forests, z-score thresholds, and autoencoder-based detectors all evaluate a stock by its own historical behavior. They are useful, but they miss a fundamental property of equity markets: stocks are correlated, and the most informative anomalies are often stocks that behave unusually _relative to their correlated peers_.

A Graph Neural Network changes the frame of reference. Instead of asking "is this stock unusual compared to its own history?", it asks "is this stock unusual compared to what its graph neighbors would predict?". A banking stock that suddenly diverges from all other banking stocks is anomalous in the graph sense even if its standalone statistics look unremarkable.

The architecture used in this recipe is a Graph Variational Autoencoder. The encoder compresses each node's features into a latent distribution using two GCN layers. The decoder reconstructs the original features from a sample drawn from that distribution. A node with high reconstruction error is one the model could not explain given its graph context. That reconstruction error becomes the anomaly score.

## Cell 3: Node Feature Engineering

Each stock is represented as a node with an 11-dimensional feature vector. These features are designed to capture everything a human analyst would check when screening for unusual stock behavior: return distribution shape, volume behavior, momentum, and volatility regime.

```python
def compute_features(price_df: pd.DataFrame,
                     volume_df: pd.DataFrame,
                     window: int = 20) -> pd.DataFrame:
    """
    Compute an 11-dimensional feature vector for each stock.
    Each stock becomes one row in the output DataFrame.
    """
    returns  = price_df.pct_change().dropna()
    features = {}

    for ticker in price_df.columns:
        ret   = returns[ticker].dropna()
        vol   = volume_df[ticker].dropna()
        price = price_df[ticker].dropna()

        # Return distribution shape
        mean_ret = ret.mean()
        std_ret  = ret.std()
        skewness = ret.skew()
        kurtosis = ret.kurtosis()

        # Extreme return days
        z_scores  = (ret - ret.mean()) / (ret.std() + 1e-8)
        max_z     = z_scores.abs().max()
        n_extreme = (z_scores.abs() > 2.5).sum()

        # Volume anomaly features
        vol_z      = (vol - vol.mean()) / (vol.std() + 1e-8)
        max_vol_z  = vol_z.abs().max()
        n_vol_spike = (vol_z.abs() > 2.5).sum()

        # Trend and momentum
        sma_20       = price.rolling(window).mean().iloc[-1]
        price_vs_sma = (price.iloc[-1] / sma_20 - 1) if sma_20 > 0 else 0

        # Volatility regime
        recent_vol = ret.tail(window).std()
        hist_vol   = ret.std()
        vol_ratio  = recent_vol / (hist_vol + 1e-8)

        # Risk-adjusted return
        sharpe = mean_ret / (std_ret + 1e-8) * np.sqrt(252)

        features[ticker] = {
            "mean_return":       mean_ret,
            "volatility":        std_ret,
            "skewness":          skewness,
            "kurtosis":          kurtosis,
            "max_return_zscore": max_z,
            "n_extreme_days":    n_extreme,
            "max_volume_zscore": max_vol_z,
            "n_volume_spikes":   n_vol_spike,
            "price_vs_sma20":    price_vs_sma,
            "vol_regime_ratio":  vol_ratio,
            "sharpe_ratio":      sharpe,
        }

    feat_df = pd.DataFrame(features).T
    feat_df = feat_df.fillna(feat_df.median())
    return feat_df


print("Computing node features...")
feat_df = compute_features(price_df, volume_df)

# Standardize before passing to the GNN

print(f"Feature matrix shape: {feat_df.shape}")
print(feat_df.describe().round(4))
```

## Cell 4: Building the Correlation Graph

The graph is built from pairwise Pearson correlations of daily returns. Two stocks are connected by an edge if their return correlation exceeds `CORRELATION_THRESHOLD`. The graph is undirected. Edges carry the correlation value as a weight.

```python
def build_correlation_graph(price_df: pd.DataFrame,
                            threshold: float = 0.6) -> tuple:
    """
    Build an undirected weighted graph from return correlations.

    Nodes : one per stock
    Edges : pairs with |correlation| >= threshold
    Weights: correlation value

    If no edges exist at the given threshold, recursively retries
    with threshold * 0.8 until at least one edge is created.
    """
    returns     = price_df.pct_change().dropna()
    corr_matrix = returns.corr()
    tickers     = corr_matrix.columns.tolist()

    G           = nx.Graph()
    G.add_nodes_from(range(len(tickers)))
    edge_index  = []
    edge_weight = []

    for i in range(len(tickers)):
        for j in range(i + 1, len(tickers)):
            corr = corr_matrix.iloc[i, j]
            if abs(corr) >= threshold:
                G.add_edge(i, j, weight=corr)
                # Both directions for undirected PyG convention
                edge_index.append([i, j])
                edge_index.append([j, i])
                edge_weight.extend([corr, corr])

    if not edge_index:
        # Graph is empty at this threshold; lower it and retry
        return build_correlation_graph(price_df, threshold * 0.8)

    edge_index_t  = torch.tensor(edge_index,  dtype=torch.long).t().contiguous()
    edge_weight_t = torch.tensor(edge_weight, dtype=torch.float)

    print(f"Graph: {G.number_of_nodes()} nodes, {G.number_of_edges()} edges "
          f"(threshold={threshold:.2f})")
    print(f"Density: {nx.density(G):.4f} | "
          f"Avg degree: {np.mean([d for _, d in G.degree()]):.2f}")

    return edge_index_t, edge_weight_t, corr_matrix, G


print("Building correlation graph...")
edge_index, edge_weight, corr_matrix, nx_graph = build_correlation_graph(
    price_df, threshold=CORRELATION_THRESHOLD
)

# Wrap in a PyTorch Geometric Data object
graph_data = Data(
    x          = feat_tensor,
    edge_index = edge_index,
    edge_attr  = edge_weight,
    num_nodes  = len(active_tickers)
).to(DEVICE)

print(f"\nPyG graph data: {graph_data}")
```

The recursive fallback matters in practice. When running on a smaller universe or a period with unusual market conditions, it is possible for the default threshold to produce a disconnected or near-empty graph. The fallback ensures training always has a meaningful graph structure to work with.

## Cell 5: Defining the GVAE Model

The model has three classes: `GNNEncoder`, `GNNDecoder`, and `AnomalyGNN` which composes them.

### Encoder

The encoder takes node features and the graph structure as input. Two GCN layers with batch normalization and ELU activations produce a hidden representation. Two additional GCN layers produce the mean (`mu`) and log-standard-deviation (`logstd`) of the latent distribution for each node.

```python
class GNNEncoder(nn.Module):
    """
    Two-layer GCN encoder producing a variational latent distribution.
    Output: (mu, logstd) each of shape (num_nodes, latent_dim)
    """
    def __init__(self, in_dim: int, hidden_dim: int, latent_dim: int):
        super().__init__()
        self.conv1       = GCNConv(in_dim,     hidden_dim)
        self.conv2       = GCNConv(hidden_dim, hidden_dim)
        self.conv_mu     = GCNConv(hidden_dim, latent_dim)
        self.conv_logstd = GCNConv(hidden_dim, latent_dim)
        self.bn1         = nn.BatchNorm1d(hidden_dim)
        self.bn2         = nn.BatchNorm1d(hidden_dim)

    def forward(self, x, edge_index):
        h = F.elu(self.bn1(self.conv1(x, edge_index)))
        h = F.dropout(h, p=0.2, training=self.training)
        h = F.elu(self.bn2(self.conv2(h, edge_index)))
        return self.conv_mu(h, edge_index), self.conv_logstd(h, edge_index)
```

### Decoder

The decoder is a plain MLP. It takes a latent sample `z` for each node and reconstructs the original 11-dimensional feature vector. The reconstruction error between the original features and this reconstruction is what becomes the anomaly score.

```python
class GNNDecoder(nn.Module):
    """
    MLP decoder reconstructing node features from latent embeddings.
    High reconstruction error on a node signals anomalous behavior.
    """
    def __init__(self, latent_dim: int, hidden_dim: int, out_dim: int):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(latent_dim, hidden_dim),
            nn.ELU(),
            nn.BatchNorm1d(hidden_dim),
            nn.Dropout(0.2),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ELU(),
            nn.Linear(hidden_dim, out_dim),
        )

    def forward(self, z):
        return self.net(z)
```

### Full GVAE

`AnomalyGNN` wires the encoder and decoder together. The `reparameterize` method implements the reparameterization trick: during training, a random sample is drawn from the latent distribution using `z = mu + eps * exp(logstd)`. At inference, the mean `mu` is used directly.

```python
class AnomalyGNN(nn.Module):
    """
    Graph Variational Autoencoder for node-level anomaly detection.

    Training objective: minimize reconstruction loss + KL divergence
                        + link prediction loss.
    Anomaly score     : per-node mean squared reconstruction error.
    """
    def __init__(self, in_dim, hidden_dim, latent_dim):
        super().__init__()
        self.encoder = GNNEncoder(in_dim, hidden_dim, latent_dim)
        self.decoder = GNNDecoder(latent_dim, hidden_dim, in_dim)

    def reparameterize(self, mu, logstd):
        if self.training:
            std = logstd.exp()
            eps = torch.randn_like(std)
            return mu + eps * std
        return mu   # deterministic at inference time

    def forward(self, x, edge_index):
        mu, logstd = self.encoder(x, edge_index)
        z          = self.reparameterize(mu, logstd)
        x_recon    = self.decoder(z)
        return x_recon, mu, logstd, z

    def loss(self, x, x_recon, mu, logstd, edge_index):
        # Term 1: reconstruction accuracy
        recon_loss = F.mse_loss(x_recon, x, reduction="mean")

        # Term 2: KL divergence from standard normal prior
        kl_loss = -0.5 * torch.mean(
            1 + 2 * logstd - mu.pow(2) - (2 * logstd).exp()
        )

        # Term 3: link prediction (should edges exist between these nodes?)
        neg_edge    = negative_sampling(edge_index, num_nodes=x.size(0))
        pos_score   = (mu[edge_index[0]] * mu[edge_index[1]]).sum(dim=1)
        neg_score   = (mu[neg_edge[0]]   * mu[neg_edge[1]]).sum(dim=1)
        link_labels = torch.cat([
            torch.ones(pos_score.size(0)),
            torch.zeros(neg_score.size(0))
        ]).to(DEVICE)
        link_scores = torch.cat([pos_score, neg_score])
        link_loss   = F.binary_cross_entropy_with_logits(link_scores, link_labels)

        total_loss = recon_loss + 0.5 * kl_loss + 0.3 * link_loss
        return total_loss, recon_loss
```

The three loss terms serve different purposes. Reconstruction loss teaches the model to accurately reproduce node features. KL divergence regularizes the latent space to prevent posterior collapse. Link prediction loss anchors the embeddings to the graph structure: similar nodes in the graph should be similar in embedding space.
