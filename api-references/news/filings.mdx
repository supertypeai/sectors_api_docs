### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors Financial API
  key.
</ParamField>

### Query Parameters

<ParamField query="sector" type="string">
  Exact match on the company's sector (e.g. `financials`,
  `communication-services`).
</ParamField> <ParamField query="sub_sector" type="string">
  Exact match on the company's subsector.
  <AccordionGroup>
    <Accordion title="Get the available sub_sector">
      Get the available `sub_sector` from the
      [Subsectors](../helper-list/subsectors) endpoint.
    </Accordion>
    <Accordion title="Examples of the sub_sector">
      `banks`, `financing-service`, `insurance`, `retailing`, `tobacco`
    </Accordion>
    <Accordion title="Format">
      The API endpoint only accepts `sub_sector` in the **kebab case** format
      (lowercase and separated by "-").
    </Accordion>
  </AccordionGroup>
</ParamField> <ParamField query="tags" type="string">
  Comma‑separated list of tags associated with the filing (e.g. `Bullish,
  Investment, insider-trading`).
  <AccordionGroup>
    <Accordion title="Get the available filing tags">
      Get the available `tags` from the [News Tags](../helper-list/tags) endpoint.
    </Accordion>
    <Accordion title="Examples of the tags">
      `acquisition`, `allocation-service`, `ai`, `b2b`, `automotive`
    </Accordion>
    <Accordion title="Format">
      The API endpoint only accepts `sub_sector` in the **kebab case** format
      (lowercase and separated by "-").
    </Accordion>
  </AccordionGroup>
</ParamField> <ParamField query="tickers" type="string">
  Comma‑separated list of normalized IDX ticker symbols (e.g. `BBCA.JK,
  TLKM.JK`).
</ParamField> <ParamField query="transaction_type" type="string">
  Filter by transaction type: `buy` or `sell`.
</ParamField> <ParamField query="start" type="string">
  ISO date (YYYY‑MM‑DD) for the start of the filing timestamp filter range.
</ParamField> <ParamField query="end" type="string">
  ISO date (YYYY‑MM‑DD) for the end of the filing timestamp filter range.
</ParamField> <ParamField query="n" type="integer">
  Return the top N most recent articles, up to a maximum of 30. If a value for N
  isn't provided, it will default to 20. This setting overrides any pagination.
</ParamField> <ParamField query="page" type="integer">
  Page number for paginated results (default is 1).
</ParamField> <ParamField query="page_size" type="integer">
  Number of items per page (default defined in `CustomNewsPagination`).
</ParamField>

### Response

<ResponseField name="[]" type="Idx Filing">
  <Expandable title="Idx Filing properties">
    <ResponseField name="id" type="integer">
      Unique filing identifier.
    </ResponseField>
    <ResponseField name="created_at" type="string">
      Record creation datetime in ISO 8601 format (UTC).
    </ResponseField>
    <ResponseField name="title" type="string">
      Headline description of the filing.
    </ResponseField>
    <ResponseField name="body" type="string">
      Detailed summary of the transaction.
    </ResponseField>
    <ResponseField name="source" type="string">
      URL to the official filing PDF.
    </ResponseField>
    <ResponseField name="timestamp" type="string">
      Filing publication datetime in ISO 8601 format (UTC).
    </ResponseField>
    <ResponseField name="sector" type="string">
      Sector classification (e.g. `basic-materials`, `healthcare`).
    </ResponseField>
    <ResponseField name="sub_sector" type="string">
      Subsector classification (e.g. `basic-materials`,
      `healthcare-equipment-providers`).
    </ResponseField>
    <ResponseField name="tags" type="string[]">
      List of topical tags (e.g. `["Bullish", "Investment",
      "insider-trading"]`).
    </ResponseField>
    <ResponseField name="tickers" type="string[]">
      List of relevant IDX ticker symbols (e.g. `["SQMI.JK"]`).
    </ResponseField>
    <ResponseField name="transaction_type" type="string">
      Transaction direction: `buy` or `sell`.
    </ResponseField>
    <ResponseField name="holding_before" type="integer">
      Number of shares held prior to this transaction.
    </ResponseField>
    <ResponseField name="holding_after" type="integer">
      Number of shares held after this transaction.
    </ResponseField>
    <ResponseField name="amount_transaction" type="integer">
      Number of shares transacted.
    </ResponseField>
    <ResponseField name="holder_type" type="string">
      Category of the holder (e.g. `insider`).
    </ResponseField>
    <ResponseField name="holder_name" type="string">
      Name of the individual or entity executing the transaction.
    </ResponseField>
    <ResponseField name="price" type="string">
      Average price per share in IDR for the transaction.
    </ResponseField>
    <ResponseField name="transaction_value" type="string">
      Total value of the transaction in IDR.
    </ResponseField>
    <ResponseField name="price_transaction" type="object">
      Breakdown of prices and share counts within the transaction:
      <Expandable title="Price Transaction details">
        <ResponseField name="prices" type="integer[]">
          List of price tiers used (e.g. `[43, 48, 42]`).
        </ResponseField>
        <ResponseField name="amount_transacted" type="integer[]">
          Corresponding number of shares at each price tier.
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="share_percentage_before" type="number">
      Percentage of total shares held before the transaction.
    </ResponseField>
    <ResponseField name="share_percentage_after" type="number">
      Percentage of total shares held after the transaction.
    </ResponseField>
    <ResponseField name="share_percentage_transaction" type="number">
      Percentage of total shares represented by this transaction.
    </ResponseField>
    <ResponseField name="uid" type="string|null">
      Unique identifier field (may be `null` if not assigned).
    </ResponseField>
  </Expandable>
</ResponseField>
