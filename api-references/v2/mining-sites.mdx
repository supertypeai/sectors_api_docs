### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors Financial API
  key.
</ParamField>

### Query Parameters

<ParamField query="province" type="string">
  Filter mining sites by province name (e.g., `Kalimantan Timur`).
</ParamField>
<ParamField query="mineral_type" type="string">
  Filter sites by the primary type of mineral (e.g., `Coal`).
</ParamField>
<ParamField query="min_production" type="number">
  Filter for sites with a production volume greater than or equal to this value.
</ParamField>
<ParamField query="sort_by" type="string" default="year">
  Sort results by `production` (descending) or `year` (descending).
</ParamField>
<ParamField query="limit" type="integer" default="20">
  Number of results to return. Maximum value is 30.
</ParamField>
<ParamField query="offset" type="integer" default="0">
  Number of results to skip. Used for pagination.
</ParamField>

### Response

<ResponseField name="[]" type="MiningSite">
  <Expandable title="Mining Site properties">
    <ResponseField name="id" type="integer">
      Unique identifier for the mining site.
    </ResponseField>
    <ResponseField name="name" type="string">
      The name of the mining site.
    </ResponseField>
    <ResponseField name="year" type="integer">
      The year the data was recorded.
    </ResponseField>
    <ResponseField name="mineral_type" type="string">
      The primary mineral extracted at the site.
    </ResponseField>
    <ResponseField name="company_id" type="integer">
      The ID of the company operating the site.
    </ResponseField>
    <ResponseField name="production_volume" type="number | null">
      The volume of production for the given year.
    </ResponseField>
  </Expandable>
</ResponseField>
