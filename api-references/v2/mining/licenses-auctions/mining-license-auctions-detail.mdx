`GET /v2/mining/license-auctions/{wiup_code}/`

### Path Parameters

<ParamField path="wiup_code" type="string" required>
  The unique identifier for the auction.
</ParamField>

### Response Fields

<ResponseField name="commodity_type" type="string">
  The commodity (e.g., "Coal", "Nickel").
</ResponseField>
<ResponseField name="city" type="string">
  City or regency of the auction area.
</ResponseField>
<ResponseField name="province" type="string">
  Province of the auction area.
</ResponseField>
<ResponseField name="company_name" type="string | null">
  The name of the winning company.
</ResponseField>
<ResponseField name="winner_date" type="string | null">
  The date the auction was won (YYYY-MM-DD).
</ResponseField>
<ResponseField name="licensed_area_ha" type="float">
  Size of the area in hectares.
</ResponseField>
<ResponseField name="license_number" type="string | null">
  Official license number if assigned.
</ResponseField>
<ResponseField name="area_type" type="string">
  The type of mining area (e.g., "WIUP", "WIUPK").
</ResponseField>
<ResponseField name="kdi" type="string | null">
  The KDI code for the area.
</ResponseField>
<ResponseField name="wiup_code" type="string">
  Unique identifier for the auction.
</ResponseField>
<ResponseField name="auction_status" type="string">
  The current status of the auction process.
</ResponseField>
<ResponseField name="participant_count" type="integer">
  Total number of participants.
</ResponseField>
<ResponseField name="winner" type="boolean">
  Whether a winner has been finalized.
</ResponseField>
<ResponseField name="created_at" type="string">
  Timestamp when the record was created.
</ResponseField>
<ResponseField name="last_modified" type="string">
  Timestamp of the last update to the record.
</ResponseField>
<ResponseField name="phases" type="object[]">
  Timeline of the auction phases.
  <Expandable title="Phase properties">
    <ResponseField name="order" type="integer">
      The sequence order of the phase.
    </ResponseField>
    <ResponseField name="description" type="string">
      Description of the auction phase.
    </ResponseField>
    <ResponseField name="start_date" type="string">
      The start date of the phase (YYYY-MM-DD).
    </ResponseField>
    <ResponseField name="end_date" type="string | null">
      The end date of the phase (YYYY-MM-DD).
    </ResponseField>
  </Expandable>
</ResponseField>
<ResponseField name="participants" type="object[]">
  List of companies that participated in the auction.
  <Expandable title="Participant properties">
    <ResponseField name="NIB" type="string">
      Nomor Induk Berusaha (Business Identification Number).
    </ResponseField>
    <ResponseField name="company_name" type="string">
      Name of the participating company.
    </ResponseField>
    <ResponseField name="email" type="string">
      Contact email of the participant.
    </ResponseField>
    <ResponseField name="qualification_result" type="string">
      Qualification status (e.g., "Lolos", "Tidak Lolos").
    </ResponseField>
  </Expandable>
</ResponseField>
<ResponseField name="company_slug" type="string | null">
  The slug of the winning company.
</ResponseField>
