### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors Financial API
  key.
</ParamField>

### Query Parameters

<ParamField query="company" type="string">
  Filter by company slug.
</ParamField>
<ParamField query="province" type="string">
  Filter licenses by the province they are registered in. Exact match.
</ParamField>
<ParamField query="commodity_type" type="string">
  Filter licenses by a specific commodity (e.g., `Coal`, `Nickel`).
  Case-insensitive.
</ParamField>
<ParamField query="license_type" type="string">
  Filter by license type (e.g., `IUP`, `IUPK`). Case-insensitive.
</ParamField>
<ParamField query="activity" type="string">
  Filter by activity stage (e.g., `Eksplorasi`, `Operasi Produksi`).
  Case-insensitive.
</ParamField>
<ParamField query="cnc" type="string">
  Filter by Clear & Clean status value. Case-insensitive.
</ParamField>
<ParamField query="expiring_soon" type="boolean">
  Set to `true` to find licenses that will expire within the next 365 days.
</ParamField>
<ParamField query="order_by" type="string" default="license_expiry_date">
  Sort field. Allowed: `license_expiry_date`, `license_effective_date`,
  `licensed_area_ha`, `commodity_type`. Prefix with `-` for descending.
</ParamField>
<ParamField query="limit" type="integer" default="20">
  Number of results to return. Maximum value is 30.
</ParamField>
<ParamField query="offset" type="integer" default="0">
  Number of results to skip. Used for pagination.
</ParamField>
