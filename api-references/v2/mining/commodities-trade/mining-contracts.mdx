---
title: "List Mining Contracts"
api: "GET /v2/mining/contracts/"
description: "Lists mining contracts between mine owners and contractors. Returns all known contracts by default; filter by either party's slug."
---

<ResponseExample>
```json 200
[
    {
        "mine_owner_slug": "adaro-energy",
        "mine_owner_name": "PT Adaro Energy Indonesia Tbk",
        "contractor_slug": "saptaindra-sejati",
        "contractor_name": "PT Saptaindra Sejati",
        "contract_period_end": "2025-12-31"
    },
    {
        "mine_owner_slug": "bukit-asam",
        "mine_owner_name": "PT Bukit Asam Tbk",
        "contractor_slug": "united-tractors",
        "contractor_name": "PT United Tractors Tbk",
        "contract_period_end": "2026-06-30"
    }
]
```

```json 404
{
  "error": "No contracts found for the given filters."
}
```

</ResponseExample>

### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors Financial API
  key.
</ParamField>

### Query Parameters

<ParamField query="mine_owner" type="string">
  Filter by mine owner company slug.
</ParamField>
<ParamField query="contractor" type="string">
  Filter by contractor company slug.
</ParamField>

### Response

<ResponseField name="[]" type="Contract">
  <Expandable title="Contract properties">
    <ResponseField name="mine_owner_slug" type="string">
      Slug of the mine owner company.
    </ResponseField>
    <ResponseField name="mine_owner_name" type="string">
      Full name of the mine owner company.
    </ResponseField>
    <ResponseField name="contractor_slug" type="string">
      Slug of the contractor company.
    </ResponseField>
    <ResponseField name="contractor_name" type="string">
      Full name of the contractor company.
    </ResponseField>
    <ResponseField name="contract_period_end" type="string">
      The date when the contract period ends (YYYY-MM-DD).
    </ResponseField>
  </Expandable>
</ResponseField>
