---
title: "Company Ownership"
api: "GET /v2/mining/companies/ownership/{slug}/"
description: "Returns the ownership structure for a company identified by slug. Shows parent companies (who owns this company) and subsidiaries (what this company owns)."
---

<ResponseExample>
```json 200
{
    "slug": "adaro-energy",
    "parents": [
        {
            "name": "PT Adaro Strategic Investments",
            "slug": "adaro-strategic-investments",
            "idx_ticker": null,
            "percentage_ownership": 43.91
        }
    ],
    "subsidiaries": [
        {
            "name": "PT Adaro Minerals Indonesia Tbk",
            "slug": "adaro-minerals-indonesia",
            "idx_ticker": "ADMR.JK",
            "percentage_ownership": 83.26
        },
        {
            "name": "PT Adaro Andalan Indonesia Tbk",
            "slug": "adaro-andalan-indonesia",
            "idx_ticker": "AADI.JK",
            "percentage_ownership": 99.99
        }
    ]
}
```

```json 404
{
  "error": "Company not found."
}
```

</ResponseExample>

### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors Financial API
  key.
</ParamField>

### Path Parameters

<ParamField path="slug" type="string" required>
  The unique identifier (slug) of the company.
</ParamField>

### Response

<ResponseField name="slug" type="string">
  The requested company slug.
</ResponseField>

<ResponseField name="parents" type="OwnershipEntry[]">
  <Expandable title="Ownership Entry properties">
    <ResponseField name="name" type="string">
      Name of the parent company.
    </ResponseField>
    <ResponseField name="slug" type="string">
      Slug of the parent company.
    </ResponseField>
    <ResponseField name="idx_ticker" type="string | null">
      IDX ticker of the parent company, if applicable.
    </ResponseField>
    <ResponseField name="percentage_ownership" type="float">
      The percentage of ownership (0-100).
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="subsidiaries" type="OwnershipEntry[]">
  <Expandable title="Ownership Entry properties">
    <ResponseField name="name" type="string">
      Name of the subsidiary company.
    </ResponseField>
    <ResponseField name="slug" type="string">
      Slug of the subsidiary company.
    </ResponseField>
    <ResponseField name="idx_ticker" type="string | null">
      IDX ticker of the subsidiary company, if applicable.
    </ResponseField>
    <ResponseField name="percentage_ownership" type="float">
      The percentage of ownership (0-100).
    </ResponseField>
  </Expandable>
</ResponseField>
