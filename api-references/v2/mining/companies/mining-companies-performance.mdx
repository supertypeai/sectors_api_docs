---
title: "Mining Company Performance"
api: "GET /v2/mining/companies/performance/{slug}/"
description: "Retrieves performance data (production, sales, resources) for a specific mining company for a single year (defaults to latest)."
---

<ResponseExample>
```json 200
{
    "year": 2023,
    "available_years": [2022, 2023],
    "data": [
        {
            "year": 2023,
            "commodity_type": "Coal",
            "commodity_sub_type": "Sub-bituminous & Metallurgical Coal",
            "commodity_stats": {
                "unit": "Mt",
                "production_volume": 49.1,
                "sales_volume": 48.5,
                "overburden_removal_volume": 200.0,
                "strip_ratio": 4.07,
                "resources_reserves": [],
                "product": []
            }
        }
    ]
}
```

```json 404
{
  "error": "Company not found."
}
```

</ResponseExample>

### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors Financial API
  key.
</ParamField>

### Path Parameters

<ParamField path="slug" type="string" required>
  The unique identifier (slug) of the mining company.
</ParamField>

### Query Parameters

<ParamField query="year" type="integer">
  The year to retrieve (e.g., 2023). Defaults to the latest available year.
</ParamField>
<ParamField query="commodity_type" type="string">
  Filter by commodity (e.g., 'Nickel'). Case-insensitive.
</ParamField>

### Response

<ResponseField name="year" type="integer">
  The specific year returned for this record.
</ResponseField>

<ResponseField name="available_years" type="integer[]">
  A list of all years where performance data is available for this company.
</ResponseField>

<ResponseField name="data" type="PerformanceRecord[]">
  <Expandable title="Performance Record properties">
    <ResponseField name="year" type="integer">
      The year the performance data was recorded for.
    </ResponseField>
    <ResponseField name="commodity_type" type="string">
      The general type of commodity.
    </ResponseField>
    <ResponseField name="commodity_sub_type" type="string">
      The specific sub-type of the commodity (e.g., "Thermal Coal").
    </ResponseField>
    <ResponseField name="commodity_stats" type="object">
      A JSON object containing detailed statistics.
      <Expandable title="Commodity Stats Structure">
        <ResponseField name="unit" type="string">
          Unit of measurement (e.g., "Mt", "kg").
        </ResponseField>
        <ResponseField name="production_volume" type="number | null">
          Volume of production.
        </ResponseField>
        <ResponseField name="sales_volume" type="number | null">
          Volume of sales.
        </ResponseField>
        <ResponseField name="overburden_removal_volume" type="number | null">
          Volume of overburden removed.
        </ResponseField>
        <ResponseField name="strip_ratio" type="number | null">
          Calculated strip ratio.
        </ResponseField>
        <ResponseField name="resources_reserves" type="object[]">
          Array of resource and reserve data structures.
        </ResponseField>
        <ResponseField name="product" type="object[]">
          Array of specific product specifications/grades.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
