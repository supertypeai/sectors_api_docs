---
title: "Mining Company Detail"
api: "GET /v2/mining/companies/{slug}/"
description: "Retrieves a comprehensive overview of a single mining company by its slug, including operational details, contact info, licenses, and associated counts."
---

<ResponseExample>

  "error": "Company not found. Please provide a valid company slug."

### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors Financial API
  key.
</ParamField>

### Path Parameters

<ParamField path="slug" type="string" required>
  The unique identifier (slug) of the mining company.
</ParamField>

### Response

<ResponseField name="name" type="string">
  The full name of the company.
</ResponseField>
<ResponseField name="slug" type="string">
  The unique URL-friendly identifier for the company.
</ResponseField>
<ResponseField name="idx_ticker" type="string | null">
  The company's IDX ticker symbol.
</ResponseField>
<ResponseField name="company_type" type="string">
  The legal or functional type of the company.
</ResponseField>
<ResponseField name="operation_province" type="string">
  Province of the main operation office.
</ResponseField>
<ResponseField name="operation_district" type="string">
  District/City of the main operation office.
</ResponseField>
<ResponseField name="key_operation" type="string">
  A brief description of the company's primary operations.
</ResponseField>
<ResponseField name="representative_address" type="string">
  Full address of the representative office.
</ResponseField>
<ResponseField name="website" type="string">
  Company's official website.
</ResponseField>
<ResponseField name="phone_number" type="string">
  Company's contact phone number.
</ResponseField>
<ResponseField name="email" type="string">
  Company's contact email address.
</ResponseField>
<ResponseField name="activities" type="string[]">
  List of the company's business activities.
</ResponseField>
<ResponseField name="commodity_type" type="string[]">
  List of commodities the company deals with.
</ResponseField>
<ResponseField name="mining_license" type="object[]">
  List of detailed mining licenses held by the company.
</ResponseField>
<ResponseField name="mining_contract" type="object[]">
  List of mining contracts associated with the company.
</ResponseField>
<ResponseField name="mining_site_count" type="integer">
  The total number of mining sites associated with this company.
</ResponseField>
