---
title: "List Mining News"
api: "GET /v2/mining/news/"
description: "Searches and lists mining-related news articles, with support for full pagination and filtering by keyword or commodity."
---

### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors Financial API
  key.
</ParamField>

### Query Parameters

<ParamField query="keyword" type="string">
  Performs a full-text search on the titles of news articles.
</ParamField>
<ParamField query="commodity_type" type="string">
  Filters news articles related to a specific commodity (exact match). Options:
  `Coal`, `Nickel`, `Copper`, `Gold`, `Bauxite`, `Tin`, `Sand, Stone, Gravel`.
</ParamField>
<ParamField query="limit" type="integer" default="20">
  Number of results to return per page. Maximum value is 30.
</ParamField>
<ParamField query="offset" type="integer" default="0">
  Number of results to skip. Used for pagination.
</ParamField>

### Response

<ResponseField name="results" type="NewsArticle[]">
  <Expandable title="News Article properties">
    <ResponseField name="title" type="string">
      The headline of the news article.
    </ResponseField>
    <ResponseField name="source" type="string">
      The original source of the news.
    </ResponseField>
    <ResponseField name="timestamp" type="string">
      The publication timestamp of the article.
    </ResponseField>
    <ResponseField name="commodity_type" type="string[]">
      List of commodities associated with the news article.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pagination" type="object">
