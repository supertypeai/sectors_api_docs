### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors Financial API
  key.
</ParamField>

### Query Parameters

<ParamField query="keyword" type="string">
  Performs a full-text search on the titles of news articles.
</ParamField>
<ParamField query="commodity" type="string">
  Filters news articles that are related to a specific commodity.
</ParamField>
<ParamField query="page" type="integer" default="1">
  The page number for paginated results.
</ParamField>
<ParamField query="page_size" type="integer" default="20">
  The number of items to return per page. Maximum value is 30.
</ParamField>

### Response

<ResponseField name="<PaginatedNewsResponse>" type="object">
  <Expandable title="Paginated Response properties">
    <ResponseField name="count" type="integer">
      The total number of news articles matching the query.
    </ResponseField>
    <ResponseField name="next" type="string | null">
      The URL for the next page of results, or `null` if this is the last page.
    </ResponseField>
    <ResponseField name="previous" type="string | null">
      The URL for the previous page of results, or `null` if this is the first
      page.
    </ResponseField>
    <ResponseField name="results" type="NewsArticle[]">
      An array of news article objects for the current page.
      <Expandable title="News Article properties">
        <ResponseField name="id" type="integer">
          Unique identifier for the news article.
        </ResponseField>
        <ResponseField name="title" type="string">
          The headline of the news article.
        </ResponseField>
        <ResponseField name="source" type="string">
          The original source of the news.
        </ResponseField>
        <ResponseField name="timestamp" type="string">
          The publication timestamp of the article.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
