### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors Financial API
  key.
</ParamField>

### Query Parameters

<ParamField query="commodity" type="string">
  Filter by commodity (e.g., 'Coal').
</ParamField>
<ParamField query="province" type="string">
  Filter by province name (e.g., 'Kalimantan Tengah').
</ParamField>
<ParamField query="status" type="string">
  Filter by auction status (e.g., 'Lelang Selesai').
</ParamField>
<ParamField query="page" type="integer" default="1">
  A page number within the paginated result set.
</ParamField>
<ParamField query="page_size" type="integer" default="20">
  Number of results to return per page. Maximum value is 30.
</ParamField>

### Response

<ResponseField name="count" type="integer">
  The total number of auctions available.
</ResponseField>
<ResponseField name="next" type="string | null">
  The URL for the next page of results, or `null` if no more pages.
</ResponseField>
<ResponseField name="previous" type="string | null">
  The URL for the previous page of results, or `null` if on the first page.
</ResponseField>
<ResponseField name="results" type="Auction[]">
  An array of mining license auction objects.
  <Expandable title="Auction properties">
    <ResponseField name="id" type="integer">
      Unique identifier for the auction.
    </ResponseField>
    <ResponseField name="commodity" type="string">
      The commodity being auctioned.
    </ResponseField>
    <ResponseField name="city" type="string">
      The city or regency of the auction location.
    </ResponseField>
    <ResponseField name="province" type="string">
      The province of the auction location.
    </ResponseField>
    <ResponseField name="company_name" type="string">
      The name of the winning company.
    </ResponseField>
    <ResponseField name="date_winner" type="string">
      The date the winner was announced, in `DD MMMM YYYY` format.
    </ResponseField>
    <ResponseField name="auction_status" type="string">
      The current status of the auction.
    </ResponseField>
    <ResponseField name="jumlah_peserta" type="integer">
      The number of participants in the auction.
    </ResponseField>
  </Expandable>
</ResponseField>
