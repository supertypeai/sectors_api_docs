### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors Financial API
  key.
</ParamField>

### Query Parameters

<ParamField query="country" type="string">
  Filter by a specific country (e.g., 'Australia').
</ParamField>
<ParamField query="commodity_type" type="string">
  Filter by a specific commodity (e.g., 'Gold', 'Nickel').
</ParamField>
<ParamField query="limit" type="integer" default="20">
  Number of results to return. Maximum value is 30.
</ParamField>
<ParamField query="offset" type="integer" default="0">
  Number of results to skip. Used for pagination.
</ParamField>

### Response

<ResponseField name="[]" type="GlobalCommodityData">
  <Expandable title="Global Commodity Data properties">
    <ResponseField name="id" type="integer">
      Unique identifier for the data entry.
    </ResponseField>
    <ResponseField name="country" type="string">
      The name of the country.
    </ResponseField>
    <ResponseField name="commodity_type" type="string">
      The type of commodity.
    </ResponseField>
    <ResponseField name="resources_reserves" type="number | null">
      The volume of resources and reserves.
    </ResponseField>
    <ResponseField name="resources_reserves_share" type="number | null">
      The country's percentage share of global resources and reserves.
    </ResponseField>
    <ResponseField name="export_import" type="string | null">
      Indicates if the country is a net exporter or importer.
    </ResponseField>
    <ResponseField name="production_volume" type="object | null">
      A JSON object representing a dictionary of production volumes by year.
    </ResponseField>
    <ResponseField name="production_share" type="object | null">
      A JSON object representing a dictionary of production shares by year.
    </ResponseField>
  </Expandable>
</ResponseField>
