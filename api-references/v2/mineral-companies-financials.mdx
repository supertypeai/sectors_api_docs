---
title: "Mineral Company Financials"
api: "GET /v2/mineral/companies/{ticker}/financials/"
description: "Retrieves financial records for a company by its IDX ticker, allowing for comparison across multiple years."
---

### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors Financial API
  key.
</ParamField>

### Path Parameters

<ParamField path="ticker" type="string" required>
  The IDX ticker symbol of the company (e.g., `ITMG.JK` or `ITMG`).
</ParamField>

### Query Parameters

<ParamField query="years" type="string">
  A comma-separated list of years to retrieve financial data for (e.g.,
  `2024,2023`). If omitted, data for all available years will be returned.
</ParamField>

### Response

<ResponseField name="[]" type="FinancialRecord">
  <Expandable title="Financial Record properties">

    <ResponseField name="idx_ticker" type="string">
      The company's IDX ticker.
    </ResponseField>
    <ResponseField name="name" type="string">
      The name of the company.
    </ResponseField>
    <ResponseField name="year" type="integer">
      The year of the financial record.
    </ResponseField>
    <ResponseField name="assets" type="number">
      Total assets for the year.
    </ResponseField>
    <ResponseField name="revenue" type="number">
      Total revenue for the year.
    </ResponseField>
    <ResponseField name="revenue_breakdown" type="object">
      JSON object detailing the sources of revenue.
    </ResponseField>
    <ResponseField name="cost_of_revenue" type="number">
      Total cost of revenue for the year.
    </ResponseField>
    <ResponseField name="cost_of_revenue_breakdown" type="object">
      JSON object detailing the components of the cost of revenue.
    </ResponseField>
    <ResponseField name="net_profit" type="number">
      Net profit for the year.
    </ResponseField>

  </Expandable>
</ResponseField>
