### Header

<ParamField header="Authorization" type="string" required>
  Authorization header that should be filled with your Sectors Financial API
  key.
</ParamField>

### Query Parameters

<ParamField query="classifications" type="string" default="all">
  Comma-separated list of classifications to rank. Defaults to all:
  `dividend_yield`, `total_dividend`, `revenue`, `earnings`, `market_cap`, `pb`,
  `pe`, `ps`.
  <Accordion title="Note">
    Each classification queried counts as one API call toward your quota.
  </Accordion>
</ParamField>

<ParamField query="year" type="integer" default="current year">
  Year for which to compute rankings (e.g. `2024`). Defaults to the current
  year.
</ParamField>

<ParamField query="n_stock" type="integer" default="5">
  Number of tickers per classification (default 5).
</ParamField>

<ParamField query="sub_sector" type="string">
  Filter by subsector (kebab‑case). Provide either this or `sub_industry`, not
  both. If both are present, `sub_industry` will override.
  <AccordionGroup>
    <Accordion title="Get the available sub_sector">
      Get the available `sub_sector` from the [Subsectors](./subsectors)
      endpoint.
    </Accordion>
    <Accordion title="Examples of the sub_sector">
      `banks`, `financing-service`, `insurance`, `retailing`, `tobacco`
    </Accordion>
    <Accordion title="Format">
      The API endpoint only accepts `sub_sector` in the **kebab case** format
      (lowercase and separated by "-").
    </Accordion>
  </AccordionGroup>
</ParamField> <ParamField query="sub_industry" type="string">
  Filter by sub‑industry (kebab‑case). Provide either this or `sub_sector`, not
  both. If both are present, `sub_industry` will override.
  <AccordionGroup>
    <Accordion title="Get the available sub_industry">
      Get the available `sub_industry` from the [Subindustries](./subindustries)
      endpoint.
    </Accordion>
    <Accordion title="Examples of the sub_industry">
      `paper`, `life-insurance`, `personal-care-products`, `software`, `gold`
    </Accordion>
    <Accordion title="Format">
      The API endpoint only accepts `sub_industry` in the **kebab case** format
      (lowercase and separated by "-").
    </Accordion>
  </AccordionGroup>
</ParamField>

<ParamField query="min_mcap_billion" type="integer" default="5000">
  Minimum market cap (in IDR billions) for companies to include (e.g. `1000` for
  IDR 1 trillion).
</ParamField>

<ParamField query="filters" type="string"> 
  Comma-separated filter expressions (e.g. `pe>10,revenue>=500000000000`). 
  Operators supported: `>`, `>=`, `<`, `<=`; fields: `dividend_yield`, `total_dividend`, `revenue`, `earnings`, `pb`, `pe`, `ps`. 
</ParamField>

<ParamField query="logic" type="string" default="and">
  Applies to `filters`. Use `and` (default) to require all conditions or `or` to
  require any.
</ParamField>

<ParamField query="include_none" type="string" default="false">
  `true` to include companies with null metrics in the ranking (sorted last),
  `false` to exclude them.
</ParamField>

### Response

<ResponseField name="<classification>" type="Array">
  <Expandable title="Entry properties">
    <ResponseField name="symbol" type="string">
      IDX ticker symbol of the company.
    </ResponseField>
    <ResponseField name="company_name" type="string">
      Full registered name of the company.
    </ResponseField>
    <ResponseField name="<classification>" type="float | integer | null">
      Metric value for the given classification in the specified year. For
      **market_cap**, this is an integer (IDR). For other classifications, this
      is a float. **Maybe null** if the company does not have data for that
      metric in the target year.
    </ResponseField>
  </Expandable>
</ResponseField>
